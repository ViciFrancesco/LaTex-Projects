\NewEnviron{multimage}[1]{
    \begin{figure}[htbp]
        \centering
        \setlength{\fboxsep}{0pt}% Elimina il padding attorno alle immagini
        \renewcommand{\arraystretch}{0}% Imposta lo spaziamento tra le righe a 0
        \edef\numimages{#1}% Numero di immagini
        \edef\colwidth{\dimexpr\linewidth/\numimages}% Larghezza di ogni colonna
        \begin{tabular}{@{} *{\numimages}{c} @{}}% Inizia una tabella con il numero di colonne specificato
            \BODY% Contenuto dell'ambiente
        \end{tabular}%
    \end{figure}%
}
